cmake_minimum_required(VERSION 3.0)
project(FFExample)
find_library(AVFORMAT NAMES avformat HINTS deps/ffmpeg/libavformat)
find_library(AVCODEC NAMES avcodec HINTS deps/ffmpeg/libavcodec)
find_library(AVUTIL NAMES avutil HINTS deps/ffmpeg/libavutil)
find_library(SWRESAMPLE NAMES swresample HINTS deps/ffmpeg/libswresample)
include_directories(./deps/ffmpeg)
add_executable(FFE FFExample.c)
target_link_libraries(FFE -s -static ${AVFORMAT} ${AVCODEC} ${AVUTIL} ${SWRESAMPLE})
